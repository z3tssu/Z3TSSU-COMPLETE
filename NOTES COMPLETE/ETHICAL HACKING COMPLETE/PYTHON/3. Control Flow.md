\# If/else statements

age = 27
if age \< 18:
print("You are too young to vote.")
elif age \>= 18 and age \< 65:
print("You are eligible to vote.")
else:
print("You are too old to vote.")

\# For loops

for i in range(10):
print(i)

\# While loops

i = 0
while i \< 10:
print(i)
i += 1

\# Break and continue

for i in range(10):
if i == 5:
break
elif i == 3:
continue
print(i)

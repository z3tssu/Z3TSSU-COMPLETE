\# Calculate mean, median, and mode of a list of numbers

import statistics
data = \[1, 3, 5, 7, 9\]
mean = statistics.mean(data)
median = statistics.median(data)
mode = statistics.mode(data)
print("Mean: " + str(mean))
print("Median: " + str(median))
print("Mode: " + str(mode))
